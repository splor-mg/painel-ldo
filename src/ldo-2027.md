```js
const ldo2027 = await FileAttachment("data/ldo2027.json").json();
```

```js
const uoOpcoes = [...new Set(ldo2027.map(d => d["Unidade Orçamentária"]).filter(Boolean))].sort();
const fonteOpcoes = [...new Set(ldo2027.map(d => String(d["Fonte"])).filter(Boolean))].sort((a,b) => +a - +b);
const receitaOpcoes = [...new Set(ldo2027.map(d => d["Descrição da Receita"]).filter(Boolean))].sort();
const metodologiaOpcoes = [...new Set(ldo2027.map(d => d["Metodologia de cálculo e premissas utilizadas"]).filter(Boolean))].sort();

const filtroUO = Inputs.select(["(Todas)", ...uoOpcoes], {label: "Unidade Orçamentária", value: "(Todas)"});
const filtroFonte = Inputs.select(["(Todas)", ...fonteOpcoes], {label: "Fonte", value: "(Todas)"});
const filtroReceita = Inputs.select(["(Todas)", ...receitaOpcoes], {label: "Classificação da Receita", value: "(Todas)"});
const busca = Inputs.text({label: "Buscar na metodologia", placeholder: "Digite uma palavra-chave..."});

const uoVal = view(filtroUO);
const fonteVal = view(filtroFonte);
const receitaVal = view(filtroReceita);
const buscaVal = view(busca);
```

```js
const dadosFiltrados = ldo2027.filter(d => {
  if (uoVal !== "(Todas)" && d["Unidade Orçamentária"] !== uoVal) return false;
  if (fonteVal !== "(Todas)" && String(d["Fonte"]) !== fonteVal) return false;
  if (receitaVal !== "(Todas)" && d["Descrição da Receita"] !== receitaVal) return false;
  if (buscaVal && !String(d["Metodologia de cálculo e premissas utilizadas"] || "").toLowerCase().includes(buscaVal.toLowerCase())) return false;
  return true;
});

const totalLDO = dadosFiltrados.reduce((s, d) => s + (d["Valor LDO"] || 0), 0);
const totalRegistros = dadosFiltrados.length;
const totalUOs = new Set(dadosFiltrados.map(d => d["Unidade Orçamentária"])).size;

// Top UOs por valor
const valorPorUO = {};
for (const d of dadosFiltrados) {
  valorPorUO[d["Unidade Orçamentária"]] = (valorPorUO[d["Unidade Orçamentária"]] || 0) + (d["Valor LDO"] || 0);
}
const topUOs = Object.entries(valorPorUO)
  .map(([uo, val]) => ({uo, val}))
  .sort((a, b) => b.val - a.val)
  .slice(0, 12);
```
